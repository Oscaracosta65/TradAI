/**
 * Retrieves the next non-blank next_draw_date, expanding #__ prefix.
 * Returns DATETIME (adds ' 00:00:00' if the source is DATE-only).
 */
function getNextDrawDate($gameId, $db, $tableName) {
    $tbl = $db->quoteName($db->replacePrefix($tableName));
    $q   = $db->getQuery(true)
        ->select($db->quoteName('next_draw_date'))
        ->from($tbl)
        ->where($db->quoteName('game_id') . ' = ' . $db->quote($gameId))
        ->where($db->quoteName('next_draw_date') . ' IS NOT NULL')
        ->where($db->quoteName('next_draw_date') . " <> " . $db->quote('0000-00-00'))
        ->where($db->quoteName('next_draw_date') . " <> " . $db->quote('0000-00-00 00:00:00'))
        ->order($db->quoteName('draw_date') . ' DESC')
        ->setLimit(1);
    $db->setQuery($q);
    $val = $db->loadResult();

    // Normalize DATE to DATETIME for inserts downstream
    if (!empty($val) && preg_match('/^\d{4}-\d{2}-\d{2}$/', $val)) {
        $val .= ' 00:00:00';
    }
    return $val ?: null;
}

// (A) Helper to compute “n choose k”
function combination($n, $k) {
    if ($k > $n) return 0;
    $res = 1;
    for ($i = 1; $i <= $k; $i++) {
        $res = $res * ($n - $i + 1) / $i;
    }
    return round($res);
}

        [[summary]]Training Parameters Explained[[/summary]]
        [[h4 style="margin-top: 10px;"]]Epochs[[/h4]]
        [[p]][[strong]]Definition:[[/strong]] An epoch is one complete pass through the entire dataset.[[/p]]
        [[p]][[strong]]Typical Range:[[/strong]] 10 to 1000.[[/p]]
        [[ul style="padding-left: 20px; margin-top: 5px;"]]
          [[li]]Increasing epochs can help the model learn more patterns but may cause overfitting.[[/li]]
          [[li]]Decreasing epochs speeds training but may underfit the data.[[/li]]
        [[/ul]]
        [[h4 style="margin-top: 10px;"]]Batch Size[[/h4]]
        [[p]][[strong]]Definition:[[/strong]] How many samples are processed at once before updating the model.[[/p]]
        [[p]][[strong]]Typical Range:[[/strong]] 8 to 256.[[/p]]
        [[ul style="padding-left: 20px; margin-top: 5px;"]]
          [[li]]Larger batch sizes speed up training but may miss finer details.[[/li]]
          [[li]]Smaller batches are more precise but slower.[[/li]]
        [[/ul]]
        [[h4 style="margin-top: 10px;"]]Dropout Rate[[/h4]]
        [[p]][[strong]]Definition:[[/strong]] The fraction of neurons randomly ignored during training to prevent overfitting.[[/p]]
        [[p]][[strong]]Typical Range:[[/strong]] 0.1 to 0.5.[[/p]]
        [[ul style="padding-left: 20px; margin-top: 5px;"]]
          [[li]]Higher dropout reduces overfitting but can slow learning.[[/li]]
          [[li]]Lower dropout allows more learning but may overfit.[[/li]]
        [[/ul]]
        [[h4 style="margin-top: 10px;"]]Learning Rate[[/h4]]
        [[p]][[strong]]Definition:[[/strong]] Controls how quickly the model adjusts its knowledge each step.[[/p]]
        [[p]][[strong]]Typical Range:[[/strong]] 0.0001 to 0.1.[[/p]]
        [[ul style="padding-left: 20px; margin-top: 5px;"]]
          [[li]]Higher rates learn faster but can skip important details.[[/li]]
          [[li]]Lower rates learn more carefully but slowly.[[/li]]
        [[/ul]]
        [[h4 style="margin-top: 10px;"]]Activation Function[[/h4]]
        [[p]][[strong]]Definition:[[/strong]] The function that decides how a neuron fires. Introduces complexity to model patterns.[[/p]]
        [[p]][[strong]]Common Choices:[[/strong]] ReLU, Sigmoid, Tanh.[[/p]]
        [[h4 style="margin-top: 10px;"]]Hidden Layers[[/h4]]
        [[p]][[strong]]Definition:[[/strong]] Layers between the input and output. More layers can detect more complex patterns.[[/p]]
        [[p]][[strong]]Typical Range:[[/strong]] 1 to 5.[[/p]]
        
        [[h4 style="margin-top: 10px;"]]Recency Decay[[/h4]]
        [[p]][[strong]]Definition:[[/strong]] Recency Decay controls how much we care about old draws versus new ones. If it’s 1.0, every draw is treated exactly the same. If it’s lower (like 0.8), each step back in time becomes a bit less important.[[/p]]
        [[p]][[strong]]Imagine this:[[/strong]] You collect a sticker each day. With 0.8 decay, yesterday’s sticker counts full, but the one from two days ago counts as 0.8×1=0.8, three days ago as 0.8×0.8=0.64, and so on.[[/p]]
        [[p]][[strong]]Why It Helps:[[/strong]] It lets the AI focus more on recent trends without forgetting the past completely.[[/p]]
        [[ul style="padding-left: 20px; margin-top: 5px;"]]
         [[li]]1.0 means “all days are equally important.”[[/li]]
         [[li]]0.9 means “each day back is 10% less important.”[[/li]]
         [[li]]0.5 makes older days fade very fast (half as important each step).[[/li]]
        [[/ul]]
        
        [[h4 style="margin-top: 10px;"]]What is Overfitting?[[/h4]]
        [[p]]Overfitting happens when the model memorizes the training data rather than learning general patterns. Using dropout and proper tuning helps prevent this.[[/p]]

    [[/h3]]
    [[p class="mdc-typography--body1" style="line-height:1.6;"]]
      The LottoExpert AI Engine uses a deep neural network, tuned through a few key settings:
    [[/p]]
    [[dl style="margin-left:16px; line-height:1.6;"]]
      [[dt]][[strong]]Epochs:[[​/strong]][[/dt]]
      [[dd]]How many full passes the model makes through the data. More epochs allow deeper learning, up to a point.[[/dd]]

      [[dt]][[strong]]Batch size:[[​/strong]][[/dt]]
      [[dd]]How many past draws the model processes at once before updating itself. This balances speed and stability.[[/dd]]

      [[dt]][[strong]]Dropout rate:[[​/strong]][[/dt]]
      [[dd]]A fraction of “turned-off” neurons during training that helps prevent overfitting and keeps the model honest.[[/dd]]

      [[dt]][[strong]]Learning rate:[[​/strong]][[/dt]]
      [[dd]]How big a step the model takes when it learns from an error. Too high can be unstable; too low slows improvement.[[/dd]]

      [[dt]][[strong]]Activation function:[[​/strong]][[/dt]]
      [[dd]]Non-linear functions (ReLU, Sigmoid, Tanh) that let the network capture more than simple straight-line trends.[[/dd]]

      [[dt]][[strong]]Hidden layers:[[​/strong]][[/dt]]
      [[dd]]Stacked layers that move from basic frequencies to more subtle co-occurrence patterns between numbers.[[/dd]]
    [[/dl]]
    [[p class="mdc-typography--caption" style="margin-top:8px; color:var(--text-muted); font-style:italic;"]]

// Utility: build a tensor of recency-decay weights
function buildSampleWeightTensor(numSamples, decay) {
  var weights = [];
  for (var i = 0; i < numSamples; i++) {
    weights.push(Math.pow(decay, (numSamples - 1) - i));
  }
  return tf.tensor1d(weights);
}

// Higher values make Skip & Hit bias stronger in the final probabilities.

var SKAI_EPS               = 1e-9;
var SKAI_SKIP_BETA_DEFAULT = 2.0;  // controls how strongly skipScores separate numbers (hoist-safe)
var SKAI_BLEND_DEFAULT     = 0.65; // Balanced-mode default AI weight λ (hoist-safe)

// Mode → λ (AI weight). Higher λ = more AI-forward.
function getSkaiBlendLambda(modeKey) {
  if (modeKey === 'ai-forward')     { return 0.90; }
  if (modeKey === 'conservative')   { return 0.80; }
  if (modeKey === 'explorative')    { return 0.45; }
  if (modeKey === 'skip-dominant')  { return 0.35; }
  // mixed is adaptive; computed from ai/skip distributions (see getSkaiLambda below)
  if (modeKey === 'mixed')          { return NaN; }
  // balanced / unknown
  return SKAI_BLEND_DEFAULT;
}

// Jensen–Shannon divergence for two discrete distributions (bounded, symmetric, stable).
function skaiJSDivergence(p, q) {
  var n = Math.min(p.length, q.length);
  var js = 0;
  for (var i = 0; i < n; i++) {
    var pi = Math.max(SKAI_EPS, p[i]);
    var qi = Math.max(SKAI_EPS, q[i]);
    var mi = 0.5 * (pi + qi);
    js += 0.5 * (pi * Math.log(pi / mi) + qi * Math.log(qi / mi));
  }
  // Natural log units; bounded in [0, ln(2)]
  return js;
}

// Mixed mode: λ adapts to how much AI and Skip agree.
// - If they agree (low JSD), lean AI-forward more.
// - If they disagree (high JSD), pull λ toward 0.5 to avoid over-committing.
function skaiAdaptiveLambdaMixed(aiNorm, skipProb) {
  var js = skaiJSDivergence(aiNorm, skipProb);
  var jsMax = Math.log(2); // theoretical maximum
  var agree = 1 - Math.min(1, js / jsMax); // 1=agree, 0=disagree

  // Map agreement into a safe λ band for Mixed:
  // - High agreement → λ up to ~0.80
  // - Low agreement  → λ down to ~0.50
  var lambda = 0.50 + (0.30 * agree);

  // Clamp defensively
  return Math.max(0.35, Math.min(0.90, lambda));
}

// Unified λ getter: fixed for other modes; adaptive for Mixed.
function getSkaiLambda(modeKey, aiNorm, skipProb) {
  if (modeKey === 'mixed') {
    return skaiAdaptiveLambdaMixed(aiNorm, skipProb);
  }
  var lam = getSkaiBlendLambda(modeKey);
  return (typeof lam === 'number' && isFinite(lam)) ? lam : SKAI_BLEND_DEFAULT;
}

// Mode-aware skip separation (beta). Higher beta makes Skip expert “sharper”.
function getSkaiSkipBeta(modeKey) {

  if (modeKey === 'explorative')   { return 2.5; }
  if (modeKey === 'conservative')  { return 1.6; }
  if (modeKey === 'ai-forward')    { return 1.6; }
  if (modeKey === 'mixed')         { return 2.2; }
  return SKAI_SKIP_BETA_DEFAULT; // balanced
}

// Temperature for exploration sampling (τ >= 1). τ=1 means no flattening.
function getSkaiExploreTau(modeKey) {
  if (modeKey === 'explorative') { return 1.8; }
  if (modeKey === 'mixed')       { return 1.4; }
  return 1.0;
}

// How many picks should be “explore sampled” instead of strict top-k.
function getSkaiExploreCount(modeKey, pickSize) {
  if (modeKey === 'explorative') { return Math.max(1, Math.floor(pickSize * 0.30)); }
  if (modeKey === 'mixed')       { return Math.max(1, Math.floor(pickSize * 0.20)); }
  return 0;
}

// Build a tempered mass (flattened) for sampling: p^(1/τ), normalized.
function skaiTemperedMass(probMass, tau) {
  if (!tau || tau <= 1) {
    return normalizePositive(probMass);
  }
  var raw = new Array(probMass.length);
  var inv = 1 / tau;
  for (var i = 0; i < probMass.length; i++) {
    raw[i] = Math.pow(Math.max(SKAI_EPS, probMass[i]), inv);
  }
  return normalizePositive(raw);
}

// Weighted sampling without replacement from a probability mass.
function skaiSampleWithoutReplacement(probMass, count) {
  var picks = [];
  var available = probMass.slice(0);

  for (var k = 0; k < count; k++) {
    // Build CDF
    var sum = 0;
    for (var i = 0; i < available.length; i++) {
      sum += Math.max(0, available[i]);
    }
    if (sum <= 0) {
      break;
    }

    var r = Math.random() * sum;
    var chosen = -1;
    for (var i = 0; i < available.length; i++) {
      r -= Math.max(0, available[i]);
      if (r <= 0) { chosen = i; break; }
    }
    if (chosen < 0) { break; }

    picks.push(chosen);
    available[chosen] = 0; // remove
  }

  return picks;
}

// Select final picks: exploit top-(pickSize-exploreCount), explore sample the rest.
function skaiSelectPicksFromMass(probMass, pickSize, modeKey) {
  var exploreCount = getSkaiExploreCount(modeKey, pickSize);
  var tau = getSkaiExploreTau(modeKey);

  var indices = [];
  // Exploit indices by sorting
  var ranked = probMass.map(function(p, i) { return { i: i, p: p }; }).sort(function(a, b) { return b.p - a.p; });

  var exploitCount = Math.max(0, pickSize - exploreCount);
  for (var k = 0; k < exploitCount && k < ranked.length; k++) {
    indices.push(ranked[k].i);
  }

  if (exploreCount > 0) {
    // Build remaining mass excluding already-picked exploit indices
    var remaining = probMass.slice(0);
    for (var k = 0; k < indices.length; k++) {
      remaining[indices[k]] = 0;
    }

    var tempered = skaiTemperedMass(remaining, tau);
    var sampled = skaiSampleWithoutReplacement(tempered, exploreCount);
    for (var k = 0; k < sampled.length; k++) {
      indices.push(sampled[k]);
    }
  }

  // Return sorted numeric values (1-based numbers) for stable display (dedup defensive)
  var seen = {};
  var uniq = [];
  for (var m = 0; m < indices.length; m++) {
    var idx = indices[m] + 1;
    if (!seen[idx]) { seen[idx] = true; uniq.push(idx); }
  }
  return uniq.sort(function(a, b) { return a - b; });
}

// Normalize a positive vector into sum-to-1 mass (with epsilon floor).
function normalizePositive(arr) {
  var sum = 0;
  for (var i = 0; i < arr.length; i++) {
    var v = (typeof arr[i] === 'number' && isFinite(arr[i])) ? Math.max(SKAI_EPS, arr[i]) : SKAI_EPS;
    sum += v;
  }
  var out = new Array(arr.length);
  var denom = (sum > 0) ? sum : (arr.length * SKAI_EPS);
  for (var i = 0; i < arr.length; i++) {
    var v = (typeof arr[i] === 'number' && isFinite(arr[i])) ? Math.max(SKAI_EPS, arr[i]) : SKAI_EPS;
    out[i] = v / denom;
  }
  return out;
}

// Turn skipScores (0..1) into a proper expert distribution using a softmax-like transform.
function buildSkipExpertProbs(skipScores, beta) {
  var raw = new Array(skipScores.length);
  for (var i = 0; i < skipScores.length; i++) {
    var u = (typeof skipScores[i] === 'number' && isFinite(skipScores[i])) ? skipScores[i] : 0;
    // center at 0.5 to reduce runaway bias; beta controls separation
    raw[i] = Math.exp(beta * (u - 0.5));
  }
  return normalizePositive(raw);
}

// Log-opinion pool (geometric mixture): p ∝ ai^λ * skip^(1-λ)
function logOpinionPool(aiProbs, skipProbs, lambda) {
  var out = new Array(aiProbs.length);
  for (var i = 0; i < aiProbs.length; i++) {
    var a = Math.max(SKAI_EPS, aiProbs[i]);
    var s = Math.max(SKAI_EPS, skipProbs[i]);
    out[i] = Math.exp(lambda * Math.log(a) + (1 - lambda) * Math.log(s));
  }
  return normalizePositive(out);
}

// --- SKAI backtest proper scoring rules (math-correct evaluation) ---
function skaiClipProb(p) {
  if (typeof p !== 'number' || !isFinite(p)) { return SKAI_EPS; }
  return Math.min(1 - SKAI_EPS, Math.max(SKAI_EPS, p));
}

function skaiBrierScore(probMass, actualSet, maxVal) {
  // probMass: array length maxVal, sum-to-1
  // actualSet: Set of winning numbers in 1..maxVal
  var sum = 0;
  for (var i = 0; i < maxVal; i++) {
    var p = skaiClipProb(probMass[i]);
    var y = actualSet.has(i + 1) ? 1 : 0;
    var d = p - y;
    sum += d * d;
  }
  return sum / maxVal;
}

function skaiLogLoss(probMass, actualSet, maxVal) {
  // Strictly proper for Bernoulli marginals; clipped to avoid -Inf
  var sum = 0;
  for (var i = 0; i < maxVal; i++) {
    var p = skaiClipProb(probMass[i]);
    var y = actualSet.has(i + 1) ? 1 : 0;
    sum += y ? -Math.log(p) : -Math.log(1 - p);
  }
  return sum / maxVal;
}

function skaiExpectedMassInPicks(probMass, picks) {
  // picks: array of numbers (1..max)
  var s = 0;
  for (var i = 0; i < picks.length; i++) {
    var idx = picks[i] - 1;
    if (idx >= 0 && idx < probMass.length) {
      s += skaiClipProb(probMass[idx]);
    }
  }
  return s;
}

// Compute per-draw Skip feature vectors for a chronological series of draws.
// drawsNumbers: [ [n1, n2, ...], ... ] in OLDEST → NEWEST order.
// For each draw t, we compute skipVec[t][k] = draws since number (k+1) last appeared BEFORE draw t.
function computeSkipFeatureSeries(drawsNumbers, pickSize, mainNumbersMax) {
  var lastHitIndex   = new Array(mainNumbersMax).fill(null);
  var skipsPerStep   = [];

  for (var t = 0; t < drawsNumbers.length; t++) {
    var skipVec = new Array(mainNumbersMax).fill(0);

    // For each number, how long since last seen?
    for (var k = 0; k < mainNumbersMax; k++) {
      var last = lastHitIndex[k];
      // If we've never seen it, treat skip as t (all previous draws missed it).
      var skip = (last === null) ? t : (t - last - 1);
      skipVec[k] = skip;
    }

    // Now update lastHitIndex using the CURRENT draw's main numbers.
    var draw = drawsNumbers[t];
    for (var j = 0; j < pickSize; j++) {
      var n = draw[j];
      if (n >= 1 && n <= mainNumbersMax) {
        lastHitIndex[n - 1] = t;
      }
    }

    skipsPerStep.push(skipVec);
  }

  // Normalize skips to 0..1 so they play nicely with normalized inputs.
  var maxSkip = 0;
  for (var t = 0; t < skipsPerStep.length; t++) {
    var vec = skipsPerStep[t];
    for (var i = 0; i < vec.length; i++) {
      if (vec[i] > maxSkip) {
        maxSkip = vec[i];
      }
    }
  }

  if (maxSkip > 0) {
    for (var t = 0; t < skipsPerStep.length; t++) {
      var vec = skipsPerStep[t];
      for (var i = 0; i < vec.length; i++) {
        vec[i] = vec[i] / maxSkip;
      }
    }
  }

  return skipsPerStep;
}

// Compute per-number Skip scores (0..1) from a DESC list of draws (most recent first).
// This is a thin wrapper around computeSkipFeatureSeries so older code that calls
// computeSkipScores(...) continues to work.
function computeSkipScores(drawsDesc, pickSize, mainNumbersMax) {
  // Convert DESC draws into ASC array of numbers for computeSkipFeatureSeries.
  var drawsAscNumbers = drawsDesc.slice().reverse().map(function(d) { return d.numbers; });
  if (!drawsAscNumbers.length) {
    return new Array(mainNumbersMax).fill(0);
  }

  var skipSeries = computeSkipFeatureSeries(drawsAscNumbers, pickSize, mainNumbersMax);
  var latestVec  = skipSeries[skipSeries.length - 1] || [];
  var result     = new Array(mainNumbersMax).fill(0);

  for (var i = 0; i < mainNumbersMax; i++) {
    result[i] = (typeof latestVec[i] === 'number') ? latestVec[i] : 0;
  }

  return result;
}

async function trainModelForBacktest(trainingDraws, settings, onEpochEndCallback) {
  var pickSize       = settings.PICK_SIZE;
  var mainNumbersMax = settings.MAIN_MAX;
  var extraBallMax   = settings.EXTRA_MAX;
  var hasExtraBall   = settings.HAS_EXTRA;

  // 1) normalize inputs
  var drawsNumbers = trainingDraws.map(function(d) { return d.numbers; });
  var normalized   = drawsNumbers.map(function(arr) {
    return arr.map(function(n, i) {
      return i < pickSize
        ? n / mainNumbersMax
        : (hasExtraBall ? n / extraBallMax : n);
    });
  });

  // 1b) compute Skip feature series (per draw, per main number)
  var skipFeatureSeries = computeSkipFeatureSeries(drawsNumbers, pickSize, mainNumbersMax);

  // 2) build X and Y arrays
  var xs = [], ys_main = [], ys_extra = [];
  for (var i = 0; i < normalized.length - 1; i++) {
    var baseInput    = normalized[i];
    var skipFeatures = skipFeatureSeries[i];
    var inputVec     = baseInput.concat(skipFeatures);
    xs.push(inputVec);

    var nextNums = drawsNumbers[i + 1];

    // Y for main
    var mainVec = new Array(mainNumbersMax).fill(0);
    nextNums.slice(0, pickSize).forEach(function(n) { mainVec[n - 1] = 1; });
    ys_main.push(mainVec);

    // Y for extra
    if (hasExtraBall) {
      var extraVec = new Array(extraBallMax).fill(0);
      extraVec[nextNums[pickSize] - 1] = 1;
      ys_extra.push(extraVec);
    }
  }

  // 3) create tensors
  var xsTensor     = tf.tensor2d(xs);
  var ysMainTensor = tf.tensor2d(ys_main);
  var ysExtraTensor;
  if (hasExtraBall) {
    ysExtraTensor = tf.tensor2d(ys_extra);
  }

  // 4) recency-decay weights
  var sampleWeightTensor = buildSampleWeightTensor(xs.length, settings.RECENCY_DECAY);

  // 5) build the model
  var input = tf.input({ shape: [xs[0].length] });
  var x = tf.layers.dense({ units: 128, activation: settings.ACTIVATION_FUNCTION }).apply(input);
  x = tf.layers.dropout({ rate: settings.DROPOUT_RATE }).apply(x);
  for (var i = 0; i < settings.HIDDEN_LAYERS; i++) {
    x = tf.layers.dense({ units: 64, activation: settings.ACTIVATION_FUNCTION }).apply(x);
    x = tf.layers.dropout({ rate: settings.DROPOUT_RATE }).apply(x);
  }
  var mainOut = tf.layers.dense({
    units: settings.MAIN_MAX,
    activation: 'sigmoid',
    name: 'main_numbers_output'
  }).apply(x);

  var model, outputs;
  if (hasExtraBall) {
    var extraOut = tf.layers.dense({
      units: settings.EXTRA_MAX,
      activation: 'sigmoid',
      name: 'extra_output'
    }).apply(x);
    outputs = [mainOut, extraOut];
    model = tf.model({ inputs: input, outputs: outputs });
  } else {
    outputs = mainOut;
    model = tf.model({ inputs: input, outputs: outputs });
  }

  // 6) compile
  var optimizer = tf.train.adam(settings.LEARNING_RATE);
  var loss      = hasExtraBall
    ? ['binaryCrossentropy','binaryCrossentropy']
    : 'binaryCrossentropy';
  model.compile({ optimizer: optimizer, loss: loss, metrics: ['accuracy'] });

  // 7) prepare yTrain (no sampleWeight support in this tfjs build)
  var yTrain;
  if (hasExtraBall) {
    yTrain = {
      main_numbers_output: ysMainTensor,
      extra_output: ysExtraTensor
    };
  } else {
    yTrain = ysMainTensor;
  }

  // 8) fit model (recency-weighted)
  // sampleWeightTensor is a per-sample exponential decay weight (newest samples weigh more).
  var fitArgs = {
    epochs:          settings.EPOCHS,
    batchSize:       settings.BATCH_SIZE,
    shuffle:         false,
    validationSplit: 0.2,
    callbacks:       { onEpochEnd: onEpochEndCallback }
  };

          // PATCH: sampleWeight not supported in this TFJS build; disable to avoid runtime error.
          // Recency is still used in backtesting via SKAI blend + skip features; full per-sample weighting requires a TFJS build that supports sampleWeight.
          // fitArgs.sampleWeight = hasExtraBall
          //   ? [sampleWeightTensor, sampleWeightTensor]
          //   : sampleWeightTensor;

  await model.fit(xsTensor, yTrain, fitArgs);


  return model;
}


async function runBacktesting(sortedDrawsAsc, settings) {
  var results      = [];
  var N            = sortedDrawsAsc.length;

  // Proper scoring aggregates (lower is better for Brier/LogLoss)
  var totalBrierMain = 0;
  var totalLogLossMain = 0;
  var totalExpectedMassInPicksMain = 0;

  var totalBrierExtra = 0;
  var totalLogLossExtra = 0;
  var totalExpectedMassInPicksExtra = 0;
  var numBacktests = parseInt(document.getElementById('numDrawsBacktest').value) || 3;
  var backtestIndices = [];
  for (var i = N - numBacktests; i < N; i++) if (i >= 1) backtestIndices.push(i);

  // progress tracking
  var totalSteps = backtestIndices.length * settings.EPOCHS;
  var completed    = 0;
  function updateBacktestProgress() {
    completed++;
    var pct = Math.round((completed/totalSteps)*100);
    backtestProgressBar.style.width   = pct + '%';
    backtestProgressBar.textContent   = pct + '%';
    progressMessage.innerText         = 'System Status: Backtesting in progress... (' + pct + '%)';
  }

  for (var bi = 0; bi < backtestIndices.length; bi++) {
    var idx = backtestIndices[bi];
    var nums       = sortedDrawsAsc[idx].numbers;
    var actualMain = nums.slice(0, settings.PICK_SIZE);
    var actualExtra= settings.HAS_EXTRA
      ? nums.slice(settings.PICK_SIZE, settings.PICK_SIZE + settings.EXTRA_COUNT)
      : [];

    // train & predict
    var trainingDraws = sortedDrawsAsc.slice(0, idx);
    var model         = await trainModelForBacktest(trainingDraws, settings, updateBacktestProgress);

    // prepare last input (normalized numbers + Skip features)
    var drawsNumbersForBT   = trainingDraws.map(function(d) { return d.numbers; });
    var skipFeatureSeriesBT = computeSkipFeatureSeries(drawsNumbersForBT, settings.PICK_SIZE, settings.MAIN_MAX);
    var lastIndex           = trainingDraws.length - 1;

    var lastBaseNorm = drawsNumbersForBT[lastIndex].map(function(v,i) {
      return i < settings.PICK_SIZE
        ? v / settings.MAIN_MAX
        : (settings.HAS_EXTRA ? v / settings.EXTRA_MAX : v);
    });
    var lastSkipFeatures = skipFeatureSeriesBT[lastIndex];
    var lastInputVec     = lastBaseNorm.concat(lastSkipFeatures);

    var pred = model.predict(tf.tensor2d([ lastInputVec ]));

    var mainArr  = settings.HAS_EXTRA ? pred[0].dataSync() : pred.dataSync();
    var extraArr = settings.HAS_EXTRA ? pred[1].dataSync() : [];

    // Compute Skip scores for this specific backtest window.
    // trainingDraws is ASC (oldest to newest), so we reverse to get DESC for computeSkipScores().
    var skipScoresForBacktest = computeSkipScores(
      trainingDraws.slice().reverse(),
      settings.PICK_SIZE,
      settings.MAIN_MAX
    );

    // Build full blended mass for proper scoring (main)
    var actualMainSet = new Set(actualMain);

    var aiRawMain = Array.from(mainArr).map(function(v) {
      var p = (typeof v === 'number' && isFinite(v)) ? v : 0;
      return Math.min(1 - SKAI_EPS, Math.max(SKAI_EPS, p));
    });
    var aiNormMain = normalizePositive(aiRawMain);

    var skipArrMain = Array.isArray(skipScoresForBacktest)
      ? skipScoresForBacktest
      : new Array(aiNormMain.length).fill(0);

    // Mode-aware skip expert + adaptive Mixed λ (agreement-based)
    var modeKeyBT = getSelectedSkaiMode();

    var skipBetaBT = getSkaiSkipBeta(modeKeyBT);
    var skipProbMain = buildSkipExpertProbs(skipArrMain, skipBetaBT);

    var lambdaBT = getSkaiLambda(modeKeyBT, aiNormMain, skipProbMain);

    // PATCH 7B-3B: backtest readout (shows the λ being used for this run)
    var tBT = (typeof skaiGetStoredTemp === 'function') ? skaiGetStoredTemp(modeKeyBT) : 1.0;
    skaiUpdateBlendReadout(modeKeyBT, lambdaBT, skipBetaBT, tBT);

    var probMassMainNorm = logOpinionPool(aiNormMain, skipProbMain, lambdaBT);

    // Deterministic top-20 ranking for backtest evaluation (no exploration here)
    var mainProbs = probMassMainNorm
      .map(function(p, i) {
        var skipScore = (Array.isArray(skipScoresForBacktest) && skipScoresForBacktest[i] != null)
          ? skipScoresForBacktest[i]
          : 0;

        return {
          number:    i + 1,
          prob:      p,
          baseProb:  aiNormMain[i],
          skipScore: skipScore
        };
      })
      .sort(function(a, b) { return b.prob - a.prob; })
      .slice(0, 20);
    var mainPicks = mainProbs.slice(0, settings.PICK_SIZE).map(function(o) { return o.number; });

    // Proper scoring updates (main)
    totalBrierMain += skaiBrierScore(probMassMainNorm, actualMainSet, settings.MAIN_MAX);
    totalLogLossMain += skaiLogLoss(probMassMainNorm, actualMainSet, settings.MAIN_MAX);
    totalExpectedMassInPicksMain += skaiExpectedMassInPicks(probMassMainNorm, mainPicks);

    var mainMatch=0;
    var predictedMain = mainProbs.map(function(x) {
      if (actualMain.includes(x.number)) { mainMatch++; return '[[span class="match"]]' + x.number + '[[/span]]'; }
      return x.number;
    });
    // top-extras
    var predictedExtra = [], extraMatch=0;
    if (settings.HAS_EXTRA) {
      var actualExtraSet = new Set(actualExtra);

      // Build full normalized mass for extra
      var aiRawExtra = Array.from(extraArr).map(function(v) {
        var p = (typeof v === 'number' && isFinite(v)) ? v : 0;
        return Math.min(1 - SKAI_EPS, Math.max(SKAI_EPS, p));
      });
      var probMassExtraNorm = normalizePositive(aiRawExtra);

      var extraProbs = Array.from(extraArr)
        .map(function(p,i) { return { number: i+1, prob: p }; })
        .sort(function(a,b) { return b.prob-a.prob; })
        .slice(0, settings.EXTRA_SUGGESTIONS);

      // Picks used for evaluation (top EXTRA_COUNT)
      var extraPicks = extraProbs.slice(0, settings.EXTRA_COUNT).map(function(o) { return o.number; });

      // Proper scoring updates (extra)
      totalBrierExtra += skaiBrierScore(probMassExtraNorm, actualExtraSet, settings.EXTRA_MAX);
      totalLogLossExtra += skaiLogLoss(probMassExtraNorm, actualExtraSet, settings.EXTRA_MAX);
      totalExpectedMassInPicksExtra += skaiExpectedMassInPicks(probMassExtraNorm, extraPicks);
      predictedExtra = extraProbs.map(function(x) {
        if (actualExtra.includes(x.number)) { extraMatch++; return '[[span class="match"]]' + x.number + '[[/span]]'; }
        return x.number;
      });
    }

    results.push({
      date:             sortedDrawsAsc[idx].draw_date,
      actualMain:       actualMain.join(', '),
      actualExtra:      actualExtra,
      predictedMain:    predictedMain.join(', '),
      predictedExtra:   predictedExtra,
      mainMatchCount:   mainMatch,
      extraMatchCount:  extraMatch
    });
  }

  // Attach a compact scoring summary for rendering
  var steps = results.length || 1;
  results.skaiSummary = {
    steps: steps,
    avgBrierMain: (totalBrierMain / steps),
    avgLogLossMain: (totalLogLossMain / steps),
    avgExpectedMassMain: (totalExpectedMassInPicksMain / steps),

    avgBrierExtra: (steps > 0 ? (totalBrierExtra / steps) : 0),
    avgLogLossExtra: (steps > 0 ? (totalLogLossExtra / steps) : 0),
    avgExpectedMassExtra: (steps > 0 ? (totalExpectedMassInPicksExtra / steps) : 0)
  };

  return results;
}

(function () {
  "use strict";

  function showSaveToast(msg) {
    try {
      var old = document.getElementById("skai-save-toast");
      if (old && old.parentNode) { old.parentNode.removeChild(old); }
      var t = document.createElement("div");
      t.id = "skai-save-toast";
      t.setAttribute("role", "status");
      t.setAttribute("aria-live", "polite");
      t.style.cssText = "position:fixed;bottom:28px;left:50%;transform:translateX(-50%);background:#065f46;color:#fff;padding:10px 22px;border-radius:999px;font-size:0.92rem;font-weight:600;box-shadow:0 4px 16px rgba(0,0,0,0.18);z-index:99999;pointer-events:none;opacity:1;transition:opacity 0.4s;";
      t.textContent = msg;
      document.body.appendChild(t);
      setTimeout(function () {
        try { t.style.opacity = "0"; } catch (_) {}
        setTimeout(function () { try { if (t.parentNode) { t.parentNode.removeChild(t); } } catch (_) {} }, 450);
      }, 3200);
    } catch (_) {}
  }

  function showInlineSaveStatus(form) {
    try {
      var existing = form.querySelector("[data-skai-inline-saved]");
      if (existing && existing.parentNode) { existing.parentNode.removeChild(existing); }
      var now = new Date();
      var hh = ("0" + now.getHours()).slice(-2);
      var mm = ("0" + now.getMinutes()).slice(-2);
      var ss = ("0" + now.getSeconds()).slice(-2);
      var span = document.createElement("span");
      span.setAttribute("data-skai-inline-saved", "1");
      span.style.cssText = "display:inline-block;margin-left:10px;color:#065f46;font-size:0.85rem;font-weight:600;vertical-align:middle;";
      span.textContent = "Saved at " + hh + ":" + mm + ":" + ss;
      var actions = form.querySelector(".skai-save-actions");
      if (actions) {
        actions.appendChild(span);
      } else {
        form.appendChild(span);
      }
    } catch (_) {}
  }

  function interceptSaveForm(form) {
    if (!form || form.__skaiSaveIntercepted) { return; }
    form.__skaiSaveIntercepted = true;

    form.addEventListener("submit", function (ev) {
      try { ev.preventDefault(); } catch (_) {}
      try { ev.stopPropagation(); } catch (_) {}

      var submitBtn = form.querySelector("[type=submit]");
      if (submitBtn) { submitBtn.disabled = true; }

      var hint = form.querySelector("[id$=save-hint], [id$=-save-hint]");
      if (hint) { hint.textContent = "Saving..."; }

      var data = [];
      var inputs = form.elements;
      for (var i = 0; i < inputs.length; i++) {
        var el = inputs[i];
        if (!el.name) { continue; }
        var tag = el.tagName.toUpperCase();
        if (tag === "INPUT") {
          if (el.type === "checkbox" || el.type === "radio") {
            if (!el.checked) { continue; }
          }
          data.push(encodeURIComponent(el.name) + "=" + encodeURIComponent(el.value || ""));
        } else if (tag === "TEXTAREA" || tag === "SELECT") {
          data.push(encodeURIComponent(el.name) + "=" + encodeURIComponent(el.value || ""));
        } else if (tag === "BUTTON" && el.type === "submit") {
          /* <button> elements have empty value by default; use "1" to match standard
             HTML form submission behavior where a clicked submit button sends name=1 */
          data.push(encodeURIComponent(el.name) + "=" + encodeURIComponent(el.value || "1"));
        }
      }
      var body = data.join("&");
      var url = form.getAttribute("action") || (window.location.href.split("#")[0]);

      var xhr = new XMLHttpRequest();
      xhr.open("POST", url, true);
      xhr.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
      xhr.setRequestHeader("X-Requested-With", "XMLHttpRequest");
      xhr.onreadystatechange = function () {
        if (xhr.readyState !== 4) { return; }
        if (submitBtn) { submitBtn.disabled = false; }
        var ok = (xhr.status >= 200 && xhr.status < 400);
        var resp = ok ? (xhr.responseText || "") : "";
        var looksOk = ok && (
          resp.indexOf("saved") >= 0 ||
          resp.indexOf("Saved") >= 0 ||
          resp.indexOf("dashboard") >= 0 ||
          resp.indexOf("success") >= 0 ||
          resp.length > 200
        );
        if (looksOk) {
          showSaveToast("Saved");
          showInlineSaveStatus(form);
          if (hint) { hint.textContent = "Saved successfully."; }
        } else {
          showSaveToast("Save request sent. If you do not see it later, try saving again.");
          if (hint) { hint.textContent = "Save request sent. If you do not see it later, try saving again."; }
        }
      };
      try { xhr.send(body); } catch (e) {
        if (submitBtn) { submitBtn.disabled = false; }
        showSaveToast("Save request sent. If you do not see it later, try saving again.");
      }
      return false;
    }, true);
  }

  function wireSaveForms() {
    var forms = document.querySelectorAll("form");
    for (var i = 0; i < forms.length; i++) {
      if (forms[i].querySelector("[name='save_set']")) {
        interceptSaveForm(forms[i]);
      }
    }
  }

  if (document.readyState === "loading") {
    document.addEventListener("DOMContentLoaded", wireSaveForms, {once: true});
  } else {
    wireSaveForms();
  }

  window.skaiInterceptSaveForm = interceptSaveForm;
  window.skaiShowSaveToast = showSaveToast;
})();

      saveAllSequential(rest, doneCallback, saveCount, totalCount);
      return;
    }

    try { selectProfile(r.combo.profile); } catch (_) {}
    try { selectStrategy(r.combo.strategy); } catch (_) {}

    var form = qs("savePredictionForm");
    if (!form) {
      saveAllSequential(rest, doneCallback, saveCount, totalCount);
      return;
    }

    try {
      if (r.mainNumbers && r.mainNumbers.length) {
        window.skaiSaveMainOrdered  = r.mainNumbers.slice();
        window.skaiSaveExtraOrdered = (r.extraNumbers && r.extraNumbers.length)
          ? r.extraNumbers.slice() : [];
        var mainField = document.getElementById("saveMainNumbers");
        if (mainField) { mainField.value = r.mainNumbers.join(","); }
        var extraField = document.getElementById("saveExtraBallNumbers");
        if (extraField) { extraField.value = (r.extraNumbers || []).join(","); }
      }
    } catch (_) {}

    /* Show saving progress: Saving 1 of 9, 2 of 9 ... */
    var currentSaveNum = saveCount + 1;
    if (window.skaiShowSaveToast) {
      window.skaiShowSaveToast("Saving " + currentSaveNum + " of " + totalCount + "...");
    }

    var ev = document.createEvent("Event");
    ev.initEvent("submit", true, true);
    form.dispatchEvent(ev);

    setTimeout(function () {
      saveAllSequential(rest, doneCallback, currentSaveNum, totalCount);
    }, 2500); /* 2.5-second delay between sequential saves */
  }

  function wireButtons() {
    var launchBtn = qs("skai-batch-launch");
    var modal = qs("skai-batch-modal");
    var confirmBtn = qs("skai-batch-confirm");
    var cancelBtn = qs("skai-batch-cancel");
    var stopBtn = qs("skai-batch-stop");
    var clearBtn = qs("skai-batch-clear");
    var detailClose = qs("skai-batch-detail-close");
    var saveAllBtn = qs("skai-batch-save-all");
    var saveSelBtn = qs("skai-batch-save-selected");

    if (launchBtn) {
      launchBtn.addEventListener("click", function () {
        if (modal) { modal.style.display = "flex"; }
      }, false);
    }

    if (cancelBtn) {
      cancelBtn.addEventListener("click", function () {
        if (modal) { modal.style.display = "none"; }
      }, false);
    }

    if (modal) {
      modal.addEventListener("click", function (e) {
        if (e.target === modal) { modal.style.display = "none"; }
      }, false);
    }

    if (confirmBtn) {
      confirmBtn.addEventListener("click", function () {
        hideModal();
        if (batchRunning) { return; }
        batchRunning = true;
        batchStopped = false;
        batchIndex = 0;
        batchDoneCount = 0;
        batchResults = [];
        var saveRow = qs("skai-batch-save-row");
        if (saveRow) { saveRow.style.display = "none"; }
        var stopBtn2 = qs("skai-batch-stop");
        if (stopBtn2) { stopBtn2.disabled = false; }
        var detailEl = qs("skai-batch-detail");
        if (detailEl) { detailEl.style.display = "none"; }
        resetGrid();
        showPanel();
        updateProgress();
        setCurrentLabel("Starting...");
        setTimeout(runNext, 400);
      }, false);
    }

    if (stopBtn) {
      stopBtn.addEventListener("click", function () {
        batchStopped = true;
        batchRunning = false;
        setCurrentLabel("Stopping...");
      }, false);
    }

    if (clearBtn) {
      clearBtn.addEventListener("click", function () {
        batchStopped = true;
        batchRunning = false;
        batchResults = [];
        batchDoneCount = 0;
        batchIndex = 0;
        try { sessionStorage.removeItem(STORAGE_KEY); } catch (_) {}
        resetGrid();
        updateProgress();
        setCurrentLabel("Ready");
        var detail2 = qs("skai-batch-detail");
        if (detail2) { detail2.style.display = "none"; }
        var saveRow2 = qs("skai-batch-save-row");
        if (saveRow2) { saveRow2.style.display = "none"; }
      }, false);
    }

    if (detailClose) {
      detailClose.addEventListener("click", function () {
        var d = qs("skai-batch-detail");
        if (d) { d.style.display = "none"; }
      }, false);
    }

    var grid = qs("skai-batch-grid");
    if (grid) {
      grid.addEventListener("click", function (e) {
        var t = e.target;
        if (t && t.getAttribute("data-batch-view") !== null) {
          try { e.preventDefault(); } catch (_) {}
          openDetail(parseInt(t.getAttribute("data-batch-view"), 10));
        }
      }, false);
    }

    if (saveAllBtn) {
      saveAllBtn.addEventListener("click", function () {
        if (!window.confirm("This will save each completed run one by one to your dashboard. Continue?")) { return; }
        var indices = [];
        for (var i = 0; i < batchResults.length; i++) {
          if (batchResults[i] && batchResults[i].done) { indices.push(i); }
        }
        if (indices.length === 0) {
          alert("No completed runs to save.");
          return;
        }
        var totalRuns = indices.length;
        saveAllSequential(indices, function () {
          if (window.skaiShowSaveToast) { window.skaiShowSaveToast("All " + totalRuns + " " + (totalRuns === 1 ? "run" : "runs") + " saved to your dashboard."); }
        });
      }, false);
    }

    if (saveSelBtn) {
      saveSelBtn.addEventListener("click", function () {
        var available = [];
        for (var i = 0; i < batchResults.length; i++) {
          if (batchResults[i] && batchResults[i].done) { available.push(i); }
        }
        if (available.length === 0) {
          alert("No completed runs to save.");
          return;
        }
        var labels = [];
        for (var j = 0; j < available.length; j++) {
          labels.push((j + 1) + ". " + batchResults[available[j]].combo.label);
        }
        var input = window.prompt(
          "Enter the numbers (comma-separated) of the runs you want to save:\n\n" + labels.join("\n"),
          "1"
        );
        if (!input) { return; }
        var chosen = [];
        var parts = input.split(",");
        for (var k = 0; k < parts.length; k++) {
          var n = parseInt(parts[k].trim(), 10);
          if (!isNaN(n) && n >= 1 && n <= available.length) {
            chosen.push(available[n - 1]);
          }
        }
        if (chosen.length === 0) { alert("No valid selections."); return; }
        saveAllSequential(chosen, function () {
          if (window.skaiShowSaveToast) { window.skaiShowSaveToast("Selected runs saved."); }
        });
      }, false);
    }
  }

  function init() {
    wireButtons();
    try { sessionStorage.removeItem(STORAGE_KEY); } catch (_) {}
  }

  if (document.readyState === "loading") {
    document.addEventListener("DOMContentLoaded", init, {once: true});
  } else {
    init();
  }
})();
